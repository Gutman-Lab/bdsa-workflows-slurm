# %%
def log(msg: str):
    print(f"[Neuropath-Notebook] {msg}")

def safe_float(x, default=0.0) -> float:
    try:
        v = float(x)
        if np.isnan(v) or np.isinf(v):
            return default
        return v
    except Exception:
        return default

def clip_int(val, lo, hi) -> int:
    if val is None:
        return lo
    try:
        if np.isnan(val):
            return lo
    except Exception:
        pass
    return int(np.clip(val, lo, hi))

def camel_to_tokens(s: str):
    return re.findall(r'[A-Z]?[a-z]+|[A-Z]+(?=[A-Z]|$)|\d+', s)

def split_all_tokens(s: str):
    parts = re.split(r'[_\-\.\s/\\]+', s)
    out = []
    for p in parts:
        if p:
            out.extend(camel_to_tokens(p))
    return out

def choose_excel_engine():
    try:
        import xlsxwriter   # noqa
        return "xlsxwriter"
    except Exception:
        try:
            import openpyxl  # noqa
            return "openpyxl"
        except Exception:
            return None

# 3) Region dictionary & extraction

# %%
REGION_ALIASES: Dict[str, List[str]] = {
    "frontal": ["frontal","fctx","front","pfc","orbitofrontal","ofc","sfg","mfg","ifg"],
    "temporal": ["temporal","tctx","temp","stg","mtg","itg","temporalpole","tp"],
    "parietal": ["parietal","pctx","par","ipg","spg"],
    "occipital": ["occipital","octx","occip","visual","v1","calcarine"],
    "cingulate": ["cingulate","cg","acc","pcc"],
    "insula": ["insula","ins","insular"],
    "entorhinal": ["entorhinal","erc","ent","entorhinalcortex","entorhinal_ctx"],
    "hippocampus": ["hippocampus","hippo","hc","ca1","ca2","ca3","dg","subiculum","dentate"],
    "amygdala": ["amygdala","amyg","amg"],
    "cerebellum": ["cerebellum","cb","vermis"]
}
REGION_ORDER = list(REGION_ALIASES.keys()) + ["Unknown"]

def _match_region_from_tokens(tokens_lower: List[str]) -> Optional[str]:
    for region, aliases in REGION_ALIASES.items():
        for a in aliases:
            if a in tokens_lower:
                return region
    joined = " ".join(tokens_lower)
    for region, aliases in REGION_ALIASES.items():
        if any(a in joined for a in aliases):
            return region
    return None

def extract_region(path_like: str) -> str:
    p = Path(path_like)
    parts_to_check = [p.name] + [parent.name for parent in p.parents[:3] if parent.name]
    tokens = []
    for part in parts_to_check:
        tokens.extend(split_all_tokens(part))
    tokens_lower = [t.lower() for t in tokens]
    region = _match_region_from_tokens(tokens_lower)
    return region or "Unknown"

# 4) JSON parser for *-ppc.anot

# %%
def parse_anot_file(path: Path):
    try:
        data = json.loads(path.read_text())
    except Exception as e:
        log(f"❌ Error reading {path.name}: {e}")
        return None

    attributes = data.get('attributes', {})
    stats_data = attributes.get('stats', {}) or attributes.get('ppc_stats', {}) or {}
    performance = attributes.get('performance', {})

    def g(key, default=0.0):
        return safe_float(stats_data.get(key, default), default)

    metrics = {
        'total_pixels': g('NumberTotalPixels', 0),
        'weak_positive': g('NumberWeakPositive', 0),
        'positive': g('NumberPositive', 0),
        'strong_positive': g('NumberStrongPositive', 0),
        'intensity_avg': g('IntensityAverage', 0),
        'processing_time': safe_float(performance.get('total_time', 0), 0.0),
        'tiles_per_second': safe_float(performance.get('tiles_per_second', 0), 0.0),
    }
    # Optional nested timing block
    time_stats = stats_data.get('time', {})
    if isinstance(time_stats, dict):
        metrics.update({
            'segmentation_time': safe_float(time_stats.get('total', 0.0), 0.0),
            'prediction_time': safe_float(time_stats.get('predictions', 0.0), 0.0),
            'merging_time': safe_float(time_stats.get('merging', 0.0), 0.0),
        })
    return metrics

# 5) Load data & compute derived metrics (SPPP, positivity, ratio) + ABC (NaN-safe)
# Scan input dir for *-ppc.anot and build the DataFrame
rows = []
ppc_files = sorted(INPUT_DIR.glob("*-ppc.anot"))
log(f"Found {len(ppc_files)} PPC annotation files in {INPUT_DIR}")

for ppc in ppc_files:
    base = ppc.stem.replace("-ppc", "")
    ppc_data = parse_anot_file(ppc)
    if not ppc_data:
        continue

    row = {
        "image_name": base,
        "region": extract_region(str(ppc)),
        **ppc_data
    }
    rows.append(row)

df = pd.DataFrame(rows)
if df.empty:
    log("⚠ No valid rows parsed. Check your input directory and JSON structures.")
else:
    # Derived metrics
    df["total_pixels"] = df["total_pixels"].replace(0, np.nan)
    pos_sum = df[["weak_positive","positive","strong_positive"]].sum(axis=1)
    df["sppp"] = (df["strong_positive"] / df["total_pixels"]) * 100
    df["positivity_rate"] = (pos_sum / df["total_pixels"]) * 100
    df["strong_ratio"] = np.where(pos_sum > 0, df["strong_positive"] / pos_sum * 100, 0)
    df["has_strong_positive"] = df["strong_positive"] > 0

    # ABC scoring (toy mapping; NaN-safe & clipped)
    def abc(row):
        sppp = safe_float(row.sppp, 0.0)
        pos_rate = safe_float(row.positivity_rate, 0.0)
        region = str(row.region or "Unknown").lower()

        # Braak (different scaling for medial temporal vs others)
        braak_f = (sppp / 5.0 + 1.0) if region in ["entorhinal","hippocampus"] else (sppp / 3.0 + 1.0)
        braak = clip_int(np.floor(braak_f), 0, 6)

        cerad = clip_int(np.floor(pos_rate / 10.0), 0, 3)
        thal  = clip_int(np.floor(sppp / 2.0 + 1.0), 0, 5)

        if braak >= 4 and cerad >= 2:
            level = "High"
        elif braak >= 3 and cerad >= 1:
            level = "Intermediate"
        else:
            level = "Low"
        return pd.Series({
            "braak_score": braak,
            "cerad_score": cerad,
            "thal_phase": thal,
            "abc_level": level,
            "abc_score": f"A{thal}B{braak}C{cerad}"
        })

    df = pd.concat([df, df.apply(abc, axis=1)], axis=1)
    df["region"] = pd.Categorical(df["region"], categories=REGION_ORDER, ordered=True)

    log(f"✅ Parsed {len(df)} samples across {df['region'].nunique()} regions.")
    display(df.head(3))

            scores = cross_val_score(clf, Xc, y_enc, cv=skf, scoring="accuracy", n_jobs=-1)
            print(f"Stratified {int(max_k)}-fold CV accuracy: mean={scores.mean():.3f} ± {scores.std():.3f}")
        else:
            print("CV skipped — not enough samples per class for ≥2 folds.")
